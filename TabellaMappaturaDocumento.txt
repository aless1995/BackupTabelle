USE [Fornitori]
-- Create a new table called 'MappaturaDocumento' in schema 'dbo'
-- Drop the table if it already exists
IF OBJECT_ID('dbo.MappaturaDocumento', 'U') IS NOT NULL
DROP TABLE dbo.MappaturaDocumento
GO
-- Create the table in the specified schema
CREATE TABLE dbo.MappaturaDocumento
(
   TimeStamp        date    NOT NULL   PRIMARY KEY, -- primary key column
   CodiceFornitore     [NVARCHAR](50)  NOT NULL,
   CodiceArticolo    [NVARCHAR](50)  NOT NULL,
   TipoDocumento     [NVARCHAR](50)  NOT NULL,
   constraint fk_MappaturaDocumento_Fornitore foreign key (CodiceFornitore) references Fornitore(CodiceFornitore),
   constraint fk_MappaturaDocumento_Articoli foreign key (CodiceArticolo) references Articoli(CodiceArticolo),
   constraint fk_MappaturaDocumento_TipiDocumento foreign key (TipoDocumento) references TipiDocumento(TipoDocumento),
);
GO
